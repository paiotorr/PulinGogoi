🛡️ Firestore Security Rules

Copy these rules into your Firestore → Rules tab. These include the required chat permissions so buyer/seller messages can sync between accounts. Do not paste old rules that fully block messages, chats, conversations, threads, or inbox collections.

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    function signedIn() {
      return request.auth != null;
    }
    // ── Admin UID list ───────────────────────────────────────────────
    // Paste the Firebase Auth UID(s) of your admin Google account(s) here.
    // Find your UID at: Firebase Console → Authentication → Users.
    // Without a real UID, admin edit / ban / delete from the admin panel
    // will fail with a "permission-denied" error.
    function isAdmin() {
      return signedIn() && request.auth.uid in [
        '69a7H0NUnXhWRa7PlNL0w98ZqN33'
      ];
    }
    function isChatParticipant(chatId) {
      return signedIn()
        && exists(/databases/$(database)/documents/chats/$(chatId))
        && request.auth.uid in get(/databases/$(database)/documents/chats/$(chatId)).data.participants;
    }

    function validUserCreate() {
      let d = request.resource.data;
      return d.role == 'user'
          && d.status == 'active';
    }

    function validUserUpdate() {
      let incoming = request.resource.data;
      let existing = resource.data;
      return incoming.role == existing.role
          && incoming.status == existing.status;
    }

    function listingAllowedKeys() {
      return [
        'id',
        'bookId',
        'title',
        'bookTitle',
        'subject',
        'className',
        'class',
        'board',
        'medium',
        'stream',
        'category',
        'seller',
        'sellerName',
        'userId',
        'sellerUserId',
        'ownerId',
        'state',
        'district',
        'subDistrict',
        'place',
        'school',
        'condition',
        'details',
        'description',
        'customSubject',
        'price',
        'adminPrice',
        'priceMode',
        'status',
        'createdAt',
        'updatedAt'
      ];
    }

    function validListingPublicFields() {
      let d = request.resource.data;
      return d.keys().hasOnly(listingAllowedKeys())
          && d.userId == request.auth.uid
          && d.sellerUserId == request.auth.uid
          && d.ownerId == request.auth.uid
          && d.title is string
          && d.className is string
          && d.state is string
          && d.district is string
          && d.place is string
          && d.price is number;
    }

    function listingOwnerIs(uid) {
      return resource.data.userId == uid
          || resource.data.sellerUserId == uid
          || resource.data.ownerId == uid;
    }

    function validListingUpdate() {
      let incoming = request.resource.data;
      let existing = resource.data;
      return validListingPublicFields()
          && listingOwnerIs(request.auth.uid)
          && (existing.userId == null || incoming.userId == existing.userId)
          && (existing.sellerUserId == null || incoming.sellerUserId == existing.sellerUserId)
          && (existing.ownerId == null || incoming.ownerId == existing.ownerId);
    }

    function validPublicUserFields(userId) {
      let d = request.resource.data;
      return d.keys().hasOnly(['uid', 'name', 'status', 'updatedAt'])
          && d.uid == userId
          && d.name is string
          && d.name.size() > 0
          && d.name.size() <= 80
          && d.status in ['active', 'inactive', 'banned'];
    }

    // Connection test
    match /_test/{doc} {
      allow read: if true;
      allow write: if false;
    }

    // App config. Public reads are needed by the exported app.
    // Only the admin UID can change catalog/config data.
    match /app_data/{doc} {
      allow read: if true;
      allow write: if isAdmin();
    }
    match /app_data/books/items/{itemId} {
      allow read: if true;
      allow write: if isAdmin();
    }

    // Users: private profile data is visible only to the owner and admin.
    // Owners can edit their own profile, but cannot change role/status.
    // Admin can read, edit, or delete user records from the admin panel.
    match /users/{userId} {
      allow read:   if (signedIn() && request.auth.uid == userId) || isAdmin();
      allow create: if signedIn()
                    && request.auth.uid == userId
                    && validUserCreate();
      allow update: if (signedIn()
                    && request.auth.uid == userId
                    && validUserUpdate())
                    || isAdmin();
      allow delete: if isAdmin();
    }

    // Public seller names only. This contains no email/mobile/DOB/private data.
    match /public_users/{userId} {
      allow read: if true;
      allow create, update: if validPublicUserFields(userId)
                    && (isAdmin()
                      || (signedIn()
                        && request.auth.uid == userId
                        && request.resource.data.status == 'active'));
      allow delete: if isAdmin();
    }

    // Listings: public book-sale data only. Private contact/profile fields are blocked.
    match /listings/{listingId} {
      allow read: if true;
      allow create: if signedIn()
                    && validListingPublicFields();
      allow update: if (isAdmin()
                    && request.resource.data.keys().hasOnly(listingAllowedKeys()))
                    || (signedIn() && validListingUpdate());
      allow delete: if isAdmin()
                    || (signedIn() && listingOwnerIs(request.auth.uid));
    }

    // Community places/schools. Signed-in users can contribute shared names.
    match /community_data/{doc} {
      allow read: if true;
      allow create: if signedIn()
                    && request.resource.data.createdBy == request.auth.uid;
      allow update: if (signedIn()
                    && resource.data.createdBy == request.auth.uid)
                    || isAdmin();
      allow delete: if isAdmin();
    }

    // Buyer interest logs
    match /buyer_interests/{logId} {
      allow read: if signedIn()
        && (resource.data.userId == request.auth.uid
            || resource.data.buyerUserId == request.auth.uid
            || resource.data.sellerUserId == request.auth.uid);
      allow create: if signedIn()
        && (request.resource.data.userId == request.auth.uid
            || request.resource.data.buyerUserId == request.auth.uid);
      allow update, delete: if false;
    }

    // Chats: only the buyer and seller in participants can read/write.
    match /chats/{chatId} {
      allow read: if signedIn() && request.auth.uid in resource.data.participants;
      allow create: if signedIn()
        && request.resource.data.participants is list
        && request.resource.data.participants.size() == 2
        && request.auth.uid in request.resource.data.participants
        && request.resource.data.buyerUid in request.resource.data.participants
        && request.resource.data.sellerUid in request.resource.data.participants;
      allow update: if signedIn()
        && request.auth.uid in resource.data.participants
        && request.resource.data.participants == resource.data.participants
        && request.resource.data.buyerUid == resource.data.buyerUid
        && request.resource.data.sellerUid == resource.data.sellerUid;
      allow delete: if false;

      match /messages/{messageId} {
        allow read: if isChatParticipant(chatId);
        allow create: if isChatParticipant(chatId)
          && (request.resource.data.senderUid == request.auth.uid
              || request.resource.data.senderId == request.auth.uid)
          && request.resource.data.text is string
          && request.resource.data.text.size() > 0
          && request.resource.data.text.size() <= 1000
          && (request.resource.data.createdAt is string
              || request.resource.data.createdAtMs is number);
        allow update: if isChatParticipant(chatId)
          && request.resource.data.diff(resource.data)
               .affectedKeys()
               .hasOnly(['deliveredTo', 'readBy']);
        allow delete: if isChatParticipant(chatId)
          && (resource.data.senderUid == request.auth.uid
              || resource.data.senderId == request.auth.uid);
      }
    }

    match /{document=**} {
      allow read, write: if false;
    }
  }
}
