🛡️ Firestore Security Rules

Copy these rules into your Firestore → Rules tab to allow the app to read data and users to write their own listings.

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // App config data — public read, admin write only
    match /app_data/{doc} {
      allow read: if true;
      allow write: if false;
    }
    // Users — users can create/update their own profile
    match /users/{userId} {
      allow read: if true;
      allow create, update: if request.auth == null || request.auth.uid == userId;
    }
    // Listings — anyone can read; authenticated users can write their own
    match /listings/{listingId} {
      allow read: if true;
      allow create: if true;
      allow update, delete: if request.resource.data.userId == resource.data.userId;
    }
  }
}
